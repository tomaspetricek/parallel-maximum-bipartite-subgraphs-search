17 void dfs(std::vector<bool>& visited, size_t& n_visited, int vert_from = 0) const
18 {
19 visited.at(vert_from) = true;
20 n_visited++;
21
22 for (const auto& vert_to : adj_list_[vert_from])
23 if (!visited[vert_to])
24 dfs(visited, n_visited, vert_to);
25 }

38 void add_edge(int vert_from, int vert_to)
39 {
40 if (vert_from>=n_vertices_ || vert_to>=n_vertices_)
41 throw std::out_of_range("Vertex index out of range");
42
43 adj_list_.at(vert_from).emplace_back(vert_to);
44 adj_list_.at(vert_to).emplace_back(vert_from);
45 n_edges_++;
46 }
