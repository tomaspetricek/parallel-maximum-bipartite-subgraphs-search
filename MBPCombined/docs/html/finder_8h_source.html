82 // DFS without B&B has complexity: O(3^n), where n is the number of edges.
83 // There are 3 options for each edge: without, with 1st coloring order
84 // and with 2nd coloring order.
85 void bb_dfs(state curr, explorer* explorer = nullptr)
86 {
87#pragma omp atomic update
88 recursion_called_++;
89
90 try_update_best(curr);
91
92 while (curr.edge_idx_<graph_.n_edges()) {
93 if (explorer)
94 if (!explorer->keep_exploring(curr))
95 return;
96
97 // check upper bound
98 if (!can_be_better(best_, curr))
99 return;
100
101 // update potential weight
102 curr.potential_weight_ += graph_.edge(curr.edge_idx()).weight;
103
104 select_edge(green, red, curr, explorer);
105
106 select_edge(red, green, curr, explorer);
107
108 // update index
109 curr.edge_idx_++;
110 }
111 }

113 std::vector<state> prepare_states(state init)
114 {
115 // color start vertex
116 init.vertex_color(0, red);
117
118 auto expl = std::make_unique<explorer>(explorer_);
119 // prepare states
120 bb_dfs(init, expl.get());
121 return expl->states();
122 }

124 // Coloring the starting vertex ensures that there is only one way (direction)
125 // to color the graph and therefore eliminates half of the possible solutions.
126 state find(const state& init)
127 {
128 if (!can_be_better(best_, init))
129 return best_;
130
131 std::vector<state> states = prepare_states(init);
132
133 assert((states.size()==0, "No states to search"));
134
135 // find best state
136#pragma omp parallel for
137 for (int i = 0; i<states.size(); i++) {
138 bb_dfs(states[i]);
139 }
140
141 return best_;
142 }
