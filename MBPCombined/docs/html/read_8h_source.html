14// Reads only lower triangle part of adjacency matrix.
15// Therefore it has complexity: O(n^2/2), where n is size of the matrix.
16pdp::graph::edge_list read_graph(const std::filesystem::path &path) {
17 std::ifstream file{path.string()};
18 std::string line;
19 int n_vertices{0};
20 int weight{0};
21 const int not_connected{0};
22
23 if (file.is_open()) {
24 // get n vertices
25 std::getline(file, line);
26 (std::istringstream(line)) >> n_vertices;
27
28 // create graph
29 pdp::graph::edge_list graph{n_vertices};
30
31 // add edges
32 for (int vert_from{0}; vert_from < n_vertices; vert_from++) {
33 std::getline(file, line);
34 std::istringstream ss(line);
35
36 for (int vert_to{0}; vert_to < vert_from; vert_to++) {
37 ss >> weight;
38
39 // add edge
40 if (weight != not_connected) {
41 graph.add_edge(pdp::graph::edge(vert_from, vert_to, weight));
42 }
43 }
44 }
45
46 return graph;
47 } else {
48 std::cerr << path.string() << std::endl;
49 throw std::runtime_error("Could not open file");
50 }
51}
