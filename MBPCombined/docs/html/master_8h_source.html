35 finder::state start()
36 {
37 auto init_states = prepare_states();
38
39 if (init_states.size()<world_.size()-1)
40 throw std::runtime_error("Not enough work for all slaves");

47 std::vector<finder::state> prepare_states()
48 {
49 auto root = finder::state(graph_.n_vertices(), graph_.n_edges());
50 std::vector<finder::state> states = finder_.prepare_states(root);
51 best_ = finder_.best();
52 return states;

55 void manage_slaves(const std::vector<finder::state>& init_states)
56 {
57 pdp::setting setting(graph_, slave_explorer_);
58 finder::state local_best;
59 boost::mpi::status status;
60
61 for (int i{0}; i<init_states.size()+world_.size(); i++) {
62 // start working
63 if (i<world_.size()-1) {
64 world_.send(i+1, tag::setting, setting);
65 world_.send(i+1, tag::config, pdp::config(init_states[i], best_));
66 }
67 // keep working
68 else if (i<init_states.size()) {
69 status = world_.recv(boost::mpi::any_source, tag::done, local_best);
70
71 // try update best
72 if (local_best.total_weight()>best_.total_weight())
73 best_ = local_best;
74
75 // send config
76 world_.send(status.source(), tag::config, pdp::config(init_states[i], best_));
77 }
78 // stop working
79 else if (i>init_states.size()) {
80 status = world_.recv(boost::mpi::any_source, tag::done, local_best);
81
82 // try update best
83 if (local_best.total_weight()>best_.total_weight())
84 best_ = local_best;
85
86 // stop slave
87 world_.send(status.source(), tag::stop, pdp::config());
88 }
89 }
90 }
