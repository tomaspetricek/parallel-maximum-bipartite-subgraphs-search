39 void select_edge(unsigned int idx, const graph::edge& edge)
40 {
41 if (idx>=selected_edges_.size())
42 throw std::out_of_range("Edge index out of range");
43
44 // add edge only if needed
45 if (!subgraph_connected_)
46 subgraph_.add_edge(edge.vert_from, edge.vert_to);
47
48 total_weight_ += edge.weight;
49
50 // select edge

52 n_selected_++;
53 }
